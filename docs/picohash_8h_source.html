116 /*
117  * The basic MD5 functions.
118  *
119  * F is optimized compared to its RFC 1321 definition just like in Colin
120  * Plumb's implementation.
121  */
122 #define _PICOHASH_MD5_F(x, y, z) ((z) ^ ((x) & ((y) ^ (z))))
123 #define _PICOHASH_MD5_G(x, y, z) ((y) ^ ((z) & ((x) ^ (y))))
124 #define _PICOHASH_MD5_H(x, y, z) ((x) ^ (y) ^ (z))
125 #define _PICOHASH_MD5_I(x, y, z) ((y) ^ ((x) | ~(z)))

127 /*
128  * The MD5 transformation for all four rounds.
129  */
130 #define _PICOHASH_MD5_STEP(f, a, b, c, d, x, t, s) \
131  (a) += f((b), (c), (d)) + (x) + (t); \
132  (a) = (((a) << (s)) | (((a)&0xffffffff) >> (32 - (s)))); \
133  (a) += (b);

135 /*
136  * SET reads 4 input bytes in little-endian byte order and stores them
137  * in a properly aligned word in host byte order.
138  *
139  * The check for little-endian architectures which tolerate unaligned
140  * memory accesses is just an optimization. Nothing will break if it
141  * doesn't work.
142  */
143 #if defined(__i386__) || defined(__x86_64__) || defined(__vax__)
144 #define _PICOHASH_MD5_SET(n) (*(const uint32_t *)&ptr[(n)*4])
145 #define _PICOHASH_MD5_GET(n) _PICOHASH_MD5_SET(n)
146 #else
147 #define _PICOHASH_MD5_SET(n) \
148  (ctx->block[(n)] = (uint_fast32_t)ptr[(n)*4] | ((uint_fast32_t)ptr[(n)*4 + 1] << 8) | ((uint_fast32_t)ptr[(n)*4 + 2] << 16) | \
149  ((uint_fast32_t)ptr[(n)*4 + 3] << 24))
150 #define _PICOHASH_MD5_GET(n) (ctx->block[(n)])

275 inline void _picohash_md5_update(_picohash_md5_ctx_t *ctx, const void *data, size_t size)
276 {
277  uint_fast32_t saved_lo;
278  unsigned long used, free;
279 
280  saved_lo = ctx->lo;
281  if ((ctx->lo = (saved_lo + size) & 0x1fffffff) < saved_lo)
282  ctx->hi++;
283  ctx->hi += size >> 29;
284 
285  used = saved_lo & 0x3f;
286 
287  if (used) {
288  free = 64 - used;
289 
290  if (size < free) {
291  memcpy(&ctx->buffer[used], data, size);
292  return;
293  }
294 
295  memcpy(&ctx->buffer[used], data, free);
296  data = (const unsigned char *)data + free;
297  size -= free;
298  _picohash_md5_body(ctx, ctx->buffer, 64);
299  }
300 
301  if (size >= 64) {
302  data = _picohash_md5_body(ctx, data, size & ~(unsigned long)0x3f);
303  size &= 0x3f;
304  }
305 
306  memcpy(ctx->buffer, data, size);
307 }

309 inline void _picohash_md5_final(_picohash_md5_ctx_t *ctx, void *_digest)
310 {
311  unsigned char *digest = _digest;
312  unsigned long used, free;
313 
314  used = ctx->lo & 0x3f;
315 
316  ctx->buffer[used++] = 0x80;
317 
318  free = 64 - used;
319 
320  if (free < 8) {
321  memset(&ctx->buffer[used], 0, free);
322  _picohash_md5_body(ctx, ctx->buffer, 64);
323  used = 0;
324  free = 64;
325  }
326 
327  memset(&ctx->buffer[used], 0, free - 8);
328 
329  ctx->lo <<= 3;
330  ctx->buffer[56] = ctx->lo;
331  ctx->buffer[57] = ctx->lo >> 8;
332  ctx->buffer[58] = ctx->lo >> 16;
333  ctx->buffer[59] = ctx->lo >> 24;
334  ctx->buffer[60] = ctx->hi;
335  ctx->buffer[61] = ctx->hi >> 8;
336  ctx->buffer[62] = ctx->hi >> 16;
337  ctx->buffer[63] = ctx->hi >> 24;
338 
339  _picohash_md5_body(ctx, ctx->buffer, 64);
340 
341  digest[0] = ctx->a;
342  digest[1] = ctx->a >> 8;
343  digest[2] = ctx->a >> 16;
344  digest[3] = ctx->a >> 24;
345  digest[4] = ctx->b;
346  digest[5] = ctx->b >> 8;
347  digest[6] = ctx->b >> 16;
348  digest[7] = ctx->b >> 24;
349  digest[8] = ctx->c;
350  digest[9] = ctx->c >> 8;
351  digest[10] = ctx->c >> 16;
352  digest[11] = ctx->c >> 24;
353  digest[12] = ctx->d;
354  digest[13] = ctx->d >> 8;
355  digest[14] = ctx->d >> 16;
356  digest[15] = ctx->d >> 24;
357 
358  memset(ctx, 0, sizeof(*ctx));
359 }

371 static inline void _picohash_sha1_hash_block(_picohash_sha1_ctx_t *s)
372 {
373  uint8_t i;
374  uint32_t a, b, c, d, e, t;
375 
376  a = s->state[0];
377  b = s->state[1];
378  c = s->state[2];
379  d = s->state[3];
380  e = s->state[4];
381  for (i = 0; i < 80; i++) {
382  if (i >= 16) {
383  t = s->buffer[(i + 13) & 15] ^ s->buffer[(i + 8) & 15] ^ s->buffer[(i + 2) & 15] ^ s->buffer[i & 15];
384  s->buffer[i & 15] = _picohash_sha1_rol32(t, 1);
385  }
386  if (i < 20) {
387  t = (d ^ (b & (c ^ d))) + _PICOHASH_SHA1_K0;
388  } else if (i < 40) {
389  t = (b ^ c ^ d) + _PICOHASH_SHA1_K20;
390  } else if (i < 60) {

444 inline void _picohash_sha1_final(_picohash_sha1_ctx_t *s, void *digest)
445 {
446  // Pad with 0x80 followed by 0x00 until the end of the block
447  _picohash_sha1_add_uncounted(s, 0x80);
448  while (s->bufferOffset != 56)
449  _picohash_sha1_add_uncounted(s, 0x00);
450

452  _picohash_sha1_add_uncounted(s, s->byteCount >> 53); // Shifting to multiply by 8
453  _picohash_sha1_add_uncounted(s, s->byteCount >> 45); // as SHA-1 supports bitstreams as well as
454  _picohash_sha1_add_uncounted(s, s->byteCount >> 37); // byte.

457  _picohash_sha1_add_uncounted(s, s->byteCount >> 13);
458  _picohash_sha1_add_uncounted(s, s->byteCount >> 5);

462  { // Swap byte order back
463  int i;
464  for (i = 0; i < 5; i++) {
465  s->state[i] = (((s->state[i]) << 24) & 0xff000000) | (((s->state[i]) << 8) & 0x00ff0000) |

535 static inline void _picohash_sha256_do_final(_picohash_sha256_ctx_t *ctx, void *digest, size_t len)
536 {
537  unsigned char *out = digest;
538  size_t i;
539 
540  /* increase the length of the message */
541  ctx->length += ctx->curlen * 8;
542 
543  /* append the '1' bit */
544  ctx->buf[ctx->curlen++] = (unsigned char)0x80;
545 
546  /* if the length is currently above 56 bytes we append zeros
547  * then compress. Then we can fall back to padding zeros and length
548  * encoding like normal.
549  */
550  if (ctx->curlen > 56) {
551  while (ctx->curlen < 64) {
552  ctx->buf[ctx->curlen++] = (unsigned char)0;
553  }
554  _picohash_sha256_compress(ctx, ctx->buf);
555  ctx->curlen = 0;
556  }
557 
558  /* pad upto 56 bytes of zeroes */
559  while (ctx->curlen < 56) {
560  ctx->buf[ctx->curlen++] = (unsigned char)0;
561  }
562 
563  /* store length */
564  for (i = 0; i != 8; ++i)
565  ctx->buf[56 + i] = ctx->length >> (56 - 8 * i);
566  _picohash_sha256_compress(ctx, ctx->buf);
567 
568  /* copy output */
569  for (i = 0; i != len / 4; ++i) {
570  out[i * 4] = ctx->state[i] >> 24;
571  out[i * 4 + 1] = ctx->state[i] >> 16;
572  out[i * 4 + 2] = ctx->state[i] >> 8;
573  out[i * 4 + 3] = ctx->state[i];
574  }
575 }
